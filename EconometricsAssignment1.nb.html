# Calculate the total sales of p1p2sales for each country

#Question 5 #Using aggregate find total sales of products 1 and 2 for each store. Which store has the highest total sales of these two products? Remember this store’s number (let’s call it XYZ).

# Calculate the total sales of p1sales and p2sales for each store
sales_by_store <- aggregate(cbind(p1sales, p2sales) ~ storeNum, data = store.df, sum)
print(sales_by_store)

# Find the store with the highest total sales of products 1 and 2
XYZ <- sales_by_store[which.max(rowSums(sales_by_store[, -1])), ]
print(XYZ)
NA

#Question 6 #Select a subset of store.df containing data only for the store XYZ with the maximum combined sales of products 1 and 2 and name this subset store.XYZ.df, where XYZ is the store’s number.

# Find the store with the highest total sales of products 1 and 2
store_with_highest_sales <- sales_by_store[which.max(rowSums(sales_by_store[, -1])), ]

# Extract the store number with the highest sales
highest_sales_store_number <- store_with_highest_sales$store

# Create a subset of store.df for the store with the highest sales
store.XYZ.df <- store.df[store.df$store == highest_sales_store_number, ]
